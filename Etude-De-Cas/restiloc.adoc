public List<Expertise> LesExpertisesIndispos() {
    List<Expertise> lesIndispos = new List<Expertise>();

    foreach (Expertise uneExpertise in lesExpertises) {
        if (uneExpertise.GetIndisponibilite() != null) {
            lesIndispos.Add(uneExpertise);
        }
    }

    return lesIndispos;
}

=== Question 2.5

[source,csharp]
----
public int NbIndisponibilites(string leMotif) {
    int nb = 0;

    foreach (Expertise uneExpertise in lesExpertises) {
        if (uneExpertise.GetIndisponibilite() != null &&
            uneExpertise.GetIndisponibilite().GetMotif() == leMotif) {
            nb++;
        }
    }

    return nb;
}

SELECT
    E.idExpert,
    E.nom,
    E.prenom,
    COUNT(M.idMission) AS nbVehiculesExpertises
FROM Expert E
JOIN MissionExpertise M ON E.idExpert = M.idExpert
WHERE YEAR(M.dateMission) = 2018
GROUP BY E.idExpert, E.nom, E.prenom;

JOIN MissionExpertise M ON G.idGarage = M.idGarage
